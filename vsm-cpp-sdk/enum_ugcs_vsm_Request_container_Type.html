#include <request_container.h>

enum Type
{
    NONE                    = 0x0,
    PROCESSOR               = 0x1,
    COMPLETION_CONTEXT      = 0x2,
    ANY                     = PROCESSOR | COMPLETION_CONTEXT,
    TEMPORAL                = 0x4,

};
